Quick Start Example

from langchain.llms import OpenAI from langchain.prompts import PromptTemplate from langchain.chains import LLMChain # Initialize the language model llm = OpenAI(temperature=0.7) # Create a prompt template prompt = PromptTemplate( input_variables=["topic"], template="Write a comprehensive guide about {topic}:" ) # Create a chain chain = LLMChain(llm=llm, prompt=prompt) # Run the chain result = chain.run("machine learning") print(result)

Custom Chain Development

Creating custom chains for specific use cases and complex workflows.

from langchain.chains.base import Chain from typing import Dict, List class CustomAnalysisChain(Chain): """Custom chain for document analysis""" def __init__(self, llm, **kwargs): super().__init__(**kwargs) self.llm = llm @property def input_keys(self) -> List[str]: return ["document"] @property def output_keys(self) -> List[str]: return ["analysis"] def _call(self, inputs: Dict[str, str]) -> Dict[str, str]: # Custom processing logic analysis = self.llm(f"Analyze: {inputs['document']}") return {"analysis": analysis}

Practical Examples

Document Q&A System

Complete implementation of a document question-answering system using RAG.

from langchain.document_loaders import PyPDFLoader from langchain.chains import RetrievalQA from langchain.vectorstores import FAISS # Load and process documents loader = PyPDFLoader("document.pdf") documents = loader.load() # Split documents text_splitter = CharacterTextSplitter( chunk_size=1000, chunk_overlap=200 ) texts = text_splitter.split_documents(documents) # Create vector store embeddings = OpenAIEmbeddings() vectorstore = FAISS.from_documents(texts, embeddings) # Create Q&A chain qa_chain = RetrievalQA.from_chain_type( llm=llm, chain_type="stuff", retriever=vectorstore.as_retriever(), return_source_documents=True ) # Ask questions query = "What are the main conclusions of the document?" result = qa_chain({"query": query}) print(result["result"])

Conversational Chatbot

Building a chatbot with conversation memory and context awareness.

from langchain.memory import ConversationBufferWindowMemory from langchain.chains import ConversationChain # Initialize memory with window memory = ConversationBufferWindowMemory( k=5, # Remember last 5 exchanges return_messages=True ) # Create conversation chain conversation = ConversationChain( llm=llm, memory=memory, verbose=True ) # Chat loop while True: user_input = input("You: ") if user_input.lower() in ['quit', 'exit']: break response = conversation.predict(input=user_input) print(f"Bot: {response}")

Multi-Agent Collaboration

Coordinating multiple specialized agents for complex task completion.

from langchain.agents import initialize_agent, Tool # Define specialized tools research_tool = Tool( name="Research", func=lambda x: f"Research results for: {x}", description="Conduct research on topics" ) analysis_tool = Tool( name="Analysis", func=lambda x: f"Analysis of: {x}", description="Analyze data and information" ) # Create specialized agents researcher = initialize_agent( [research_tool], llm, agent="zero-shot-react-description" ) analyst = initialize_agent( [analysis_tool], llm, agent="zero-shot-react-description" ) # Orchestrate agents def multi_agent_workflow(topic): # Step 1: Research research_result = researcher.run(f"Research {topic}") # Step 2: Analysis analysis_result = analyst.run(f"Analyze {research_result}") return analysis_result result = multi_agent_workflow("artificial intelligence trends")
